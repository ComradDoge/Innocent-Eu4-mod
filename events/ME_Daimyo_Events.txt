country_event = { #Ashikaga Succession Crisis - For Ashikaga 
	id = ME_Daimyo_Events.1
	title = ME_Daimyo_Events.1.t
	desc = ME_Daimyo_Events.1.d
	picture = DIPLOMACY_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = ASK
		has_reform = shogunate
		is_year = 1450
		NOT = { is_year = 1500 }
		exists = HSK
		exists = YMN
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	option = { #Sad
		name = ME_Daimyo_Events.1.a
		
		set_global_flag = JAP_crisis
		add_legitimacy = -10
		custom_tooltip = ME_Daimyo_Tooltips.1
		hidden_effect = {
			every_subject_country = {
				limit = {
					ai = yes
					is_subject_of_type = daimyo_vassal
					NOT = { tag = HSK }
					NOT = { tag = YMN }
				}
				country_event = { id = ME_Daimyo_Events.2 }
				set_variable = {
					which = Daimyo_Support_for_HSK
					value = 0
				}
				set_variable = {
					which = Daimyo_Support_for_YMN
					value = 0
				}
			}
		}
	}
}

country_event = { #Kyoto Kanrei
	id = ME_Daimyo_Events.3
	title = ME_Daimyo_Events.3.t
	desc = ME_Daimyo_Events.3.d
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		has_reform = daimyo
		capital_scope = {
			OR = {
				area = sanindo_area
				area = shikoku_area
				area = saigoku_area
				area = kinai_area
				area = chubu_area
				area = hokuriku_area
			}
		}
		NOT = {
			any_country = {
				has_reform = daimyo
				capital_scope = {
					OR = {
						area = sanindo_area
						area = shikoku_area
						area = saigoku_area
						area = kinai_area
						area = chubu_area
						area = hokuriku_area
					}
				}
				army_size = ROOT
				has_opinion = {
					who = ASK
					value = 0
				}
			}
		}
		NOT = { has_country_flag = JAP_kyoto_kan }
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = { #Position Gained
		name = ME_Daimyo_Events.3.a
		
		set_country_flag = JAP_kyoto_kan
		add_country_modifier = {
			name = JAP_kyoto_kanrei
			duration = -1
		}
		every_country = {
			limit = {
				NOT = { tag = ROOT }
				OR = {
					has_country_flag = JAP_kyoto_kan
					has_country_modifier = JAP_kyoto_kanrei
				}
			}
			country_event = { id = ME_Daimyo_Events.4 }
		}
	}
}

country_event = { #Onset of the War
	id = ME_Daimyo_Events.8
	title = ME_Daimyo_Events.8.t
	desc = ME_Daimyo_Events.8.d
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		OR = {
			tag = HSK
			tag = YMN
		}
		had_global_flag = {
			flag = JAP_crisis
			days = 3650
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = ME_Daimyo_Events.8.a
		
		ai_chance = { factor = 100 }
		
		if = {
			limit = {
				tag = HSK
			}
			if = {
				limit = {
					exists = YMN
				}
				add_casus_belli = {
					target = YMN
					type = cb_onin_war
					months = 60
				}
				if = {
					limit = {
						ai = yes
					}
					declare_war_with_cb = {
						who = YMN
						casus_belli = cb_onin_war 
					}
				}
			}
		}
		else = {
			if = {
				limit = {
					exists = HSK
				}
				add_casus_belli = {
					target = HSK
					type = cb_onin_war
					months = 60
				}
				if = {
					limit = {
						ai = yes
					}
					declare_war_with_cb = {
						who = HSK
						casus_belli = cb_onin_war 
					}
				}
			}
		}
		clr_global_flag = JAP_crisis #disable ability to influence
	}
}
