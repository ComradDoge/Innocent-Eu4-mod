country_event = { #Kyoto Kanrei
	id = ME_Daimyo_Events.3
	title = ME_Daimyo_Events.3.t
	desc = ME_Daimyo_Events.3.d
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		has_reform = daimyo
		capital_scope = {
			OR = {
				area = sanindo_area
				area = shikoku_area
				area = saigoku_area
				area = kinai_area
				area = chubu_area
				area = hokuriku_area
			}
		}
		NOT = {
			any_country = {
				has_reform = daimyo
				capital_scope = {
					OR = {
						area = sanindo_area
						area = shikoku_area
						area = saigoku_area
						area = kinai_area
						area = chubu_area
						area = hokuriku_area
					}
				}
				army_size = ROOT
				has_opinion = {
					who = ASK
					value = 0
				}
			}
		}
		NOT = { has_country_flag = JAP_kyoto_kan }
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = { #Position Gained
		name = ME_Daimyo_Events.3.a
		every_country = {
			limit = {
				OR = {
					has_country_flag = JAP_kyoto_kan
					has_country_modifier = JAP_kyoto_kanrei
				}
			}
			country_event = { id = ME_Daimyo_Events.4 }
		}
		set_country_flag = JAP_kyoto_kan
		add_country_modifier = {
			name = JAP_kyoto_kanrei
			duration = -1
		}
	}
}
