the_two_sicilies_mission_1 = {
	slot = 1
	generic = no
	ai = yes
	has_country_shield = yes
	potential = {
		tag = TTS
		OR = {
			was_tag = NAP
			was_tag = SIC
		}
		OR = {
			NOT = { has_dlc = "Emperor" }
			AND = {
				has_dlc = "Emperor"
				has_country_flag = use_missions_expanded_missions_instead_of_paradox
			}
		}
	}

	tts_first_mission = {
		icon = mission_assemble_an_army
		required_missions = { }
		position = 9
		trigger = {
			army_size_percentage = 1
			manpower_percentage = 0.6
			num_of_generals = 1
		}
		effect = {
			add_army_tradition = 15.0
			add_mil_power = 100
			add_manpower = 5
			add_country_modifier = {
				name = "army_enthusiasm"
				duration = 7300
			}
			if = {
				limit = {
					is_subject = yes
				}
				add_liberty_desire = 40
			}
		}
	}
}
