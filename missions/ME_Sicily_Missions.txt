	sic_gain_independence = {
		icon = mission_assemble_an_army
		required_missions = { }
		position = 1
		trigger = {
			if = {
				limit = { is_subject_of = ARA }
				army_size_percentage = 1
				num_of_generals = 1
				OR = {
					if = {
						limit = { exists = NAP }
						NAP = {
							has_opinion = {
								who = ROOT
								value = 100
							}
						}
					}
					any_country = {
						has_opinion = {
							who = ROOT
							value = 125
						}
					}
				}
			}
			else = { stability = 1 }
		}
		effect = {
			add_yearly_manpower = 5
			add_country_modifier = {
				name = nap_struggle_anew
				duration = 3650
			}
			if = {
				limit = {
					is_subject = yes
				}
				country_event = {
					id = ME_Sicily_Events.1
				}
			}
		}
	}

	sic_matter_stability = {
		icon = mission_religious
		position = 2
		required_missions = { sic_gain_independence }
		trigger = {
			is_subject = no
			sicily_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
				NOT = { unrest = 3 }
			}
			western_sicily_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
				NOT = { unrest = 3 }
			}
		}
		effect = {
			western_sicily_area = {
				if = {
					limit = { has_dlc = "Mandate of Heaven" }
					add_prosperity = 100
				}
				else = {
					random_list = {
						33 = { add_base_tax = 1 }
						33 = { add_base_production = 1 }
						33 = { add_base_manpower = 1 }
					}
				}
				add_province_modifier = {
					name = nap_legacy_ruggero
					duration = -1
				}
			}
			sicily_area = {
				if = {
					limit = { has_dlc = "Mandate of Heaven" }
					add_prosperity = 100
				}
				else = {
					random_list = {
						33 = { add_base_tax = 1 }
						33 = { add_base_production = 1 }
						33 = { add_base_manpower = 1 }
					}
				}
				add_province_modifier = {
					name = nap_legacy_ruggero
					duration = -1
				}
			}
		}
	}
