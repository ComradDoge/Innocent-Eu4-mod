ASK_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = ASK
		OR = {
			AND = {
				NOT = { has_reform = indep_daimyo }
				NOT = { has_reform = daimyo }
				NOT = { has_reform = shogunate }
			}
			has_country_flag = use_missions_expanded_missions_instead_of_paradox
		}
	}
	has_country_shield = yes
	
	ASK_the_kyoto_contribution = {
		icon = mission_high_income
		position = 1
		required_missions = { }
		trigger = {
			custom_trigger_tooltip = {
				tooltip = ASK_DAIMYO_CONTRIBUTION_EVENT_HAS_HAPPENED
				has_country_flag = JAP_daimyo_contribution_flag
			}
		}
		effect = {
			add_stability_or_adm_power = yes
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
				}
				if = {
					limit = {
						ruler_has_personality = obsessive_perfectionist_personality
					}
					remove_ruler_personality = obsessive_perfectionist_personality
				}
				add_ruler_personality = architectural_visionary_personality
			}
			else = {
				capital_scope = {				
					add_province_modifier = {
						name = ASK_daimyo_tributes
						duration = 7300
					}
				}
			}
		}
	}
	
	ASK_development_plan = {
		icon = mission_asian_city
		position = 3
		required_missions = { ASK_the_kyoto_contribution ASK_enforce_diplomatic_attendance }
		provinces_to_highlight = {
			is_capital_of = ROOT
			OR = {
				AND = {
					NOT = { has_building = temple }
					NOT = { has_building = cathedral }
				}
				AND = { 
					NOT = { has_building = marketplace }
					NOT = { has_building = trade_depot }
					NOT = { has_building = stock_exchange }
				}
			}
		}
		trigger = {
			capital_scope = {
				OR = {
					has_building = temple
					has_building = cathedral
				}
				OR = {
					has_building = marketplace
					has_building = trade_depot
					has_building = stock_exchange
				}
			}
		}
		effect = {
			capital_scope = {
				if = {
					limit = {
						NOT = { renaissance = 100 }
					}
					add_institution_embracement = {
						which = renaissance
						value = 30
					}
				}
			}
			if = {
				limit = {
					OR = {
						NOT = { has_dlc = "Mandate of Heaven" }
						capital_scope = {
							is_prosperous = yes
						}
					}
				}
				add_years_of_income = 0.33
			}
			else = {
				capital_scope = {
					add_prosperity = 100
				}
			}
		}
	}
	
	ASK_temple = {
		icon = mission_rb_trinity_college_dublin
		position = 4
		required_missions = { ASK_development_plan }
		provinces_to_highlight = {
			is_capital_of = ROOT
			NOT = { development = 35 }
		}
		trigger = {
			capital_scope = {
				development = 35
			}
		}
		effect = {
			capital_scope = {
				add_province_modifier = {
					name = ASK_golden_pavilion
					duration = -1
				}
			}
		}
	}
}

		effect = {

				limit = {

			}
			add_prestige = 10

	}
